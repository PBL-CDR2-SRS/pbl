! =====================================================================
! R-DC  -  02 - Interfaces
! Cablagem GNS3: Eth0/3 -> R-Central (WAN)  |  Eth0/1 -> SW-DC (trunk)
! =====================================================================

! Eth0/3 - Link WAN para o R-Central (10.99.0.0/30)
interface Ethernet0/3
 ip address 10.99.0.1 255.255.255.252
 no shutdown
exit

! Eth0/1 - Trunk para o SW-DC (router-on-a-stick). Interface fisica sem IP.
! SO existem as sub-interfaces .90 e .100. Propositadamente SEM sub-interface
! .60 (Convidados) nem nativa -> o R-DC nunca encaminha trafego de Convidados.
interface Ethernet0/1
 no shutdown
exit

! Eth0/1.90 - VLAN 90 Gestao DC (10.30.0.0/29)
interface Ethernet0/1.90
 encapsulation dot1Q 90
 ip address 10.30.0.1 255.255.255.248
exit

! Eth0/1.100 - VLAN 100 Servidores DC (10.30.0.8/29)
interface Ethernet0/1.100
 encapsulation dot1Q 100
 ip address 10.30.0.9 255.255.255.248
exit
