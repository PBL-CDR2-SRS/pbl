! =====================================================================
! SW-DC  —  01 · VLANs, Trunk e Gestao (SVI VLAN 90)
! AREA PROTEGIDA: so VLAN 90 (Gestao DC) e VLAN 100 (Servidores DC).
! Convidados (VLAN 60) NAO pode entrar no DC.
! Native VLAN = 999 (Parking Lot): deixar a native na VLAN 1 seria um erro
! de seguranca ainda PIOR do que desviar da convencao do enunciado
! (Convidados=native). A 999 e uma VLAN morta, cujas portas tem port-security
! e estao em shutdown -> e a opcao mais segura.
! IP de gestao do switch: 10.30.0.2.
! =====================================================================

enable
configure terminal
hostname SW-DC
no ip domain-lookup

vlan 90
 name Gestao-DC
vlan 100
 name Servidores-DC
vlan 999
 name ParkingLot
exit

! Trunk para o R-DC (router-on-a-stick).
! Native = 999 (parking, morta) e allowed SO 90,100: a VLAN 60 nao atravessa o
! trunk e nao ha sub-interface .60 no R-DC -> Convidados morre aqui.
interface Ethernet0/0
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk native vlan 999
 switchport trunk allowed vlan 90,100
exit

! Portas de acesso dos 3 servidores (VLAN 100)
interface range Ethernet0/1 - 3
 switchport mode access
 switchport access vlan 100
 spanning-tree portfast
exit

! Parking Lot: portas nao utilizadas -> VLAN 999 + shutdown
! (port-security destas portas fica em 02-port-security.txt)
! (ajustar o range as portas reais do switch no GNS3)
interface range Ethernet1/0 - 3 , Ethernet2/0 - 3 , Ethernet3/0 - 3
 switchport mode access
 switchport access vlan 999
 shutdown
exit

! Interface de gestao (SVI VLAN 90) — gerir o switch pela rede de Gestao DC
interface vlan 90
 ip address 10.30.0.2 255.255.255.248
 no shutdown
exit

! O switch e L2: usa default-gateway para sair da sua sub-rede de gestao
ip default-gateway 10.30.0.1

end
write
