! 4 · ROTAS ESTATICAS  (controladas — so destinos especificos)
!  ALTERACOES (pedido Professor):
!   - REMOVIDA a rota Null0 de black-hole do proprio /20.
!     Ja nao e precisa: sem rota default, trafego para 10.20.x.x
!     desconhecido nao tem para onde ir -> e descartado por "no route".
!     (Antes a Null0 apanhava o que a default reenviava para o R-C e
!      causaria loop. Sem default, sem loop -> Null0 redundante.)
!   - REMOVIDA a rota default 0.0.0.0/0. Routing mais controlado:
!     so encaminhamos para redes internas conhecidas.

! Datacenter (10.30.0.0/24) via R-Central
ip route 10.30.0.0 255.255.255.0 10.99.0.14

! Edificio A (10.10.0.0/20) — PRIMARIA via R-Central (AD 1)
ip route 10.10.0.0 255.255.240.0 10.99.0.14

! Edificio A — BACKUP (floating) pelo link direto, AD=200.
!  So entra na tabela se a primaria via R-C cair. Para "flutuar"
!  corretamente, a primaria /20 acima TEM de existir como rota
!  especifica (era o que a antiga default mascarava).
ip route 10.10.0.0 255.255.240.0 10.99.0.17 200

! Links WAN internos (10.99.0.0/24) via R-Central — visibilidade/
! diagnostico (traceroute, ping a interfaces WAN). Opcional mas util.
ip route 10.99.0.0 255.255.255.0 10.99.0.14

! ---------------------------------------------------------------------
!  >>> DECISAO EM ABERTO: INTERNET PARA O EDIFICIO B <<<
!  Sem rota default, os segmentos do Ed.B NAO chegam a Internet
!  (as ACLs ainda tem "permit ... any" mas ficam sem caminho de saida).
!  Se o enunciado exigir Internet no Ed.B, DESCOMENTAR a linha abaixo.
!  Continua "controlado": e a unica rota de ultimo recurso, identificada.
! ip route 0.0.0.0 0.0.0.0 10.99.0.14   ! Internet (last resort) -> R-C
! ---------------------------------------------------------------------


! =====================================================================
! 5 · DEFINICAO DAS ACLs  (extended, inbound em cada interface LAN)
! ---------------------------------------------------------------------
!  DHCP: bootps=67 (cliente->servidor) e bootpc=68 (servidor->cliente).
!        Ambos sao precisos, senao o "deny" final mata o OFFER/ACK.
!  ICMP: adicionado "permit icmp" para os servidores que cada segmento
!        ja podia aceder por HTTP (permite ping/diagnostico).
!  Servidores DC: HTTP institucional 10.30.0.10 · HTTP interno 10.30.0.11
! =====================================================================

! --- ALUNOS Ed.B (VLAN 10 · e0/1) ---
!  Podem: DHCP, HTTP institucional (+ping), impressoras, inter-edificios, Internet
ip access-list extended ACL-ALUNOS-EDB
 permit udp any any eq bootps
 permit udp any any eq bootpc
 permit tcp 10.20.0.0 0.0.0.255 host 10.30.0.10 eq www
 permit icmp 10.20.0.0 0.0.0.255 host 10.30.0.10              ! ping ao site institucional
 permit ip 10.20.0.0 0.0.0.255 10.20.1.192 0.0.0.15           ! Impressoras Ed.B
 permit ip 10.20.0.0 0.0.0.255 10.10.0.0 0.0.15.255           ! Inter-edificios (Ed.A)
 deny ip 10.20.0.0 0.0.0.255 10.0.0.0 0.255.255.255           ! Bloqueia resto interno
 permit ip 10.20.0.0 0.0.0.255 any                            ! Internet
exit

! --- PROFESSORES Ed.B (VLAN 20 · e0/2) ---
!  Podem: DHCP, HTTP institucional + interno (+ping), impressoras, inter-edificios, Internet
ip access-list extended ACL-PROF-EDB
 permit udp any any eq bootps
 permit udp any any eq bootpc
 permit tcp 10.20.1.0 0.0.0.63 host 10.30.0.10 eq www
 permit icmp 10.20.1.0 0.0.0.63 host 10.30.0.10
 permit tcp 10.20.1.0 0.0.0.63 host 10.30.0.11 eq www
 permit icmp 10.20.1.0 0.0.0.63 host 10.30.0.11
 permit ip 10.20.1.0 0.0.0.63 10.20.1.192 0.0.0.15            ! Impressoras Ed.B
 permit ip 10.20.1.0 0.0.0.63 10.10.0.0 0.0.15.255            ! Inter-edificios (Ed.A)
 deny ip 10.20.1.0 0.0.0.63 10.0.0.0 0.255.255.255
 permit ip 10.20.1.0 0.0.0.63 any
exit

! --- FINANCEIROS Ed.B (VLAN 30 · e1/1) ---
!  Podem: DHCP, HTTP institucional (+ping), impressoras, Internet
ip access-list extended ACL-FINANCEIROS-EDB
 permit udp any any eq bootps
 permit udp any any eq bootpc
 permit tcp 10.20.1.64 0.0.0.15 host 10.30.0.10 eq www
 permit icmp 10.20.1.64 0.0.0.15 host 10.30.0.10
 permit ip 10.20.1.64 0.0.0.15 10.20.1.192 0.0.0.15           ! Impressoras Ed.B
 deny ip 10.20.1.64 0.0.0.15 10.0.0.0 0.255.255.255
 permit ip 10.20.1.64 0.0.0.15 any
exit

! --- ACADEMICOS Ed.B (VLAN 40 · e1/2) ---
!  Podem: DHCP, HTTP institucional + interno (+ping), impressoras, Internet
ip access-list extended ACL-ACADEMICOS-EDB
 permit udp any any eq bootps
 permit udp any any eq bootpc
 permit tcp 10.20.1.96 0.0.0.31 host 10.30.0.10 eq www
 permit icmp 10.20.1.96 0.0.0.31 host 10.30.0.10
 permit tcp 10.20.1.96 0.0.0.31 host 10.30.0.11 eq www
 permit icmp 10.20.1.96 0.0.0.31 host 10.30.0.11
 permit ip 10.20.1.96 0.0.0.31 10.20.1.192 0.0.0.15           ! Impressoras Ed.B
 deny ip 10.20.1.96 0.0.0.31 10.0.0.0 0.255.255.255
 permit ip 10.20.1.96 0.0.0.31 any
exit

! --- CONVIDADOS Ed.B (VLAN 60 · e1/3) ---
!  Podem: SO Internet. Nada interno (nem ICMP interno).
ip access-list extended ACL-CONVIDADOS-EDB
 permit udp any any eq bootps
 permit udp any any eq bootpc
 deny ip 10.20.1.128 0.0.0.7 10.0.0.0 0.255.255.255
 permit ip 10.20.1.128 0.0.0.7 any
exit

! --- TELEFONES Ed.B (VLAN 70 · e0/3) ---
!  Podem: DHCP, inter-edificio (Ed.A, inclui ICMP via "permit ip"). Sem Internet.
ip access-list extended ACL-TELEFONES-EDB
 permit udp any any eq bootps
 permit udp any any eq bootpc
 permit ip 10.20.1.160 0.0.0.31 10.10.0.0 0.0.15.255
 deny ip any any
exit

! --- GESTAO Ed.B (VLAN 90 · e1/0) ---
!  Podem: DHCP, inter-edificio (Ed.A). Sem Internet nem servidores de conteudo.
ip access-list extended ACL-GESTAO-EDB
 permit udp any any eq bootps
 permit udp any any eq bootpc
 permit ip 10.20.1.224 0.0.0.31 10.10.0.0 0.0.15.255

exit

! --- IMPRESSORAS Ed.B (VLAN 80 · e2/0) ---
!  Podem: SO DHCP.
ip access-list extended ACL-IMPRESSORAS-EDB
 permit udp any any eq bootps
 permit udp any any eq bootpc

exit
