# ALUNOS ED.A
# Permitido: DHCP, HTTP Institucional, Impressoras Ed.A, Inter-edifícios (Ed.B) e Internet.
ip access-list extended ACL-ALUNOS-EDA
 permit udp any host 10.30.0.2 eq 67
 permit tcp 10.10.0.0 0.0.0.127 host 10.30.0.3 eq 80
 permit ip 10.10.0.0 0.0.0.127 10.10.1.0 0.0.0.15
 permit ip 10.10.0.0 0.0.0.127 10.20.0.0 0.0.15.255
 deny ip 10.10.0.0 0.0.0.127 10.0.0.0 0.255.255.255  # Bloqueia o resto do tráfego interno
 permit ip 10.10.0.0 0.0.0.127 any                   # Permite Internet
 
# PROFESSORES ED.A
# Permitido: DHCP, HTTP Inst. e Interno, Impressoras Ed.A, Inter-edifícios (Ed.B) e Internet.
ip access-list extended ACL-PROF-EDA
 permit udp any host 10.30.0.2 eq 67
 permit tcp 10.10.0.128 0.0.0.63 host 10.30.0.3 eq 80
 permit tcp 10.10.0.128 0.0.0.63 host 10.30.0.4 eq 80
 permit ip 10.10.0.128 0.0.0.63 10.10.1.0 0.0.0.15
 permit ip 10.10.0.128 0.0.0.63 10.20.0.0 0.0.15.255
 deny ip 10.10.0.128 0.0.0.63 10.0.0.0 0.255.255.255 # Bloqueia o resto do tráfego interno
 permit ip 10.10.0.128 0.0.0.63 any                  # Permite Internet

# INFORMÁTICA ED.A
# Permitido: Acesso total a tudo (Sem restrições).
ip access-list extended ACL-INFO-EDA
 permit ip 10.10.0.192 0.0.0.15 any

# TELEFONES ED.A
# Permitido: DHCP e Inter-edifícios (Ed.B) apenas.
ip access-list extended ACL-TELEFONES-EDA
 permit udp any host 10.30.0.2 eq 67
 permit ip 10.10.0.224 0.0.0.31 10.20.0.0 0.0.15.255
 deny ip any any

# IMPRESSORAS ED.A
# Permitido: DHCP apenas.
ip access-list extended ACL-IMPRESSORAS-EDA
 permit udp any host 10.30.0.2 eq 67
 deny ip any any

# Aplicação nas sub interfaces do router Ed.A, ver se baetm certo com as do Romeu antes de fazer

 ip access-group ACL-ALUNOS-EDA in
interface GigabitEthernet0/0.20
 ip access-group ACL-PROF-EDA in
interface GigabitEthernet0/0.50
 ip access-group ACL-INFO-EDA in
interface GigabitEthernet0/0.70
 ip access-group ACL-TELEFONES-EDA in
interface GigabitEthernet0/0.80
 ip access-group ACL-IMPRESSORAS-EDA in
exit
