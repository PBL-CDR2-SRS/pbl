# ALUNOS ED.A (Rede 10.10.0.0 /25)
# Permitido: DHCP, HTTP Institucional, Impressoras Ed.A, Inter-edifícios (Ed.B) e Internet.
ip access-list extended ACL-ALUNOS-EDA
 permit udp any host 10.30.0.1 eq 67
 permit tcp 10.10.0.0 0.0.0.127 host 10.30.0.3 eq 80
 permit ip 10.10.0.0 0.0.0.127 10.10.5.0 0.0.0.15
 permit ip 10.10.0.0 0.0.0.127 10.20.0.0 0.0.15.255
 deny ip 10.10.0.0 0.0.0.127 10.0.0.0 0.255.255.255  # Bloqueia o resto do tráfego interno
 permit ip 10.10.0.0 0.0.0.127 any                   # Permite Internet
 
# PROFESSORES ED.A (Rede 10.10.1.0 /26)
# Permitido: DHCP, HTTP Inst. e Interno, Impressoras Ed.A, Inter-edifícios (Ed.B) e Internet.
ip access-list extended ACL-PROF-EDA
 permit udp any host 10.30.0.1 eq 67
 permit tcp 10.10.1.0 0.0.0.63 host 10.30.0.3 eq 80
 permit tcp 10.10.1.0 0.0.0.63 host 10.30.0.4 eq 80
 permit ip 10.10.1.0 0.0.0.63 10.10.5.0 0.0.0.15
 permit ip 10.10.1.0 0.0.0.63 10.20.0.0 0.0.15.255
 deny ip 10.10.1.0 0.0.0.63 10.0.0.0 0.255.255.255 # Bloqueia o resto do tráfego interno
 permit ip 10.10.1.0 0.0.0.63 any                  # Permite Internet

# INFORMÁTICA ED.A (Rede 10.10.2.0 /29)
# Permitido: Acesso total a tudo (Sem restrições).
ip access-list extended ACL-INFO-EDA
 permit ip 10.10.2.0 0.0.0.7 any

# TELEFONES ED.A (Rede 10.10.4.0 /27)
# Permitido: DHCP e Inter-edifícios (Ed.B) apenas.
ip access-list extended ACL-TELEFONES-EDA
 permit udp any host 10.30.0.1 eq 67
 permit ip 10.10.4.0 0.0.0.31 10.20.0.0 0.0.15.255
 deny ip any any

# IMPRESSORAS ED.A (Rede 10.10.5.0 /28)
# Permitido: DHCP apenas.

 permit udp any host 10.30.0.1 eq 67

 #PERMITIR PING (ICMP) PARA TESTES DO PROJETO

 permit icmp 10.20.0.0 0.0.15.255 10.30.0.0 0.0.0.7 echo
 
 #HTTP INSTITUCIONAL (.3) -> Aberto a toda a Universidade e Internet (TCP 80)
 permit tcp any host 10.30.0.3 eq 80
 
 #HTTP INTERNO (.4) -> Restrito a Professores (A e B) e Académicos (B)
 permit tcp 10.10.1.0 0.0.0.63 host 10.30.0.4 eq 80  # Professores Ed.A
 permit tcp 10.20.1.0 0.0.0.63 host 10.30.0.4 eq 80  # Professores Ed.B
 permit tcp 10.20.3.0 0.0.0.31 host 10.30.0.4 eq 80  # Académicos Ed.B
 
 #FTP (.5) - Restrito à Informática Ed.A (Portas TCP 21 Control e 20 Data)
 permit tcp 10.10.2.0 0.0.0.7 host 10.30.0.5 eq 21
 permit tcp 10.10.2.0 0.0.0.7 host 10.30.0.5 eq 20
 
 #BLOQUEAR RESTO COM LOG DE AUDITORIA
 deny ip any any log

interface Ethernet0/0
 ip access-group ACL-WAN-IN in
exit

Problema 2: Prevenção de Routing Loops na Sumarização (Black Hole Route)

descartar tráfego destinado a IPs não utilizados dentro do bloco /20, 
evitando que o tráfego volte para trás pela rota default.
